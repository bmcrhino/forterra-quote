  function switchTab(tabId) {
    // Update tab buttons
    document.querySelectorAll('.forterra-pricing .category-tab').forEach(btn => {
      btn.classList.toggle('active', btn.dataset.tab === tabId);
    });
    // Update plan tab panels
    document.querySelectorAll('.forterra-pricing .tab-panel').forEach(panel => {
      panel.classList.toggle('active', panel.dataset.tabPanel === tabId);
    });
    // Update plan headings + CTA
    const info = tabHeadings[tabId];
    if (info) {
      const subtitle = document.getElementById('plansSubtitle');
      const heading = document.getElementById('plansHeading');
      if (subtitle) subtitle.textContent = info.subtitle;
      if (heading) heading.textContent = info.heading;
    }
    // Re-apply current billing state to new panel
    setBilling(currentBilling);
  }

  // ---- Billing Toggle ----
  let currentBilling = 'monthly';

  function setBilling(period) {
    currentBilling = period;
    const buttons = document.querySelectorAll('.forterra-pricing .billing-toggle button');
    buttons.forEach(btn => {
      btn.classList.toggle('active', btn.dataset.period === period);
    });

    // Update ALL amount elements across every tab panel
    const amounts = document.querySelectorAll('.forterra-pricing .plan-price .amount');
    const notes = document.querySelectorAll('.forterra-pricing .annual-note');

    amounts.forEach(el => {
      const monthly = parseInt(el.dataset.monthly);
      const annual = parseInt(el.dataset.annual);
      if (monthly && annual) {
        el.textContent = period === 'annual' ? annual : monthly;
      }
    });

    notes.forEach(el => {
      if (period === 'annual') {
        // Calculate annual total: monthly price × 11 months (1 month free)
        const card = el.closest('.forterra-pricing .plan-card');
        if (card) {
          const amountEl = card.querySelector('.plan-price .amount');
          if (amountEl) {
            const monthly = parseInt(amountEl.dataset.monthly);
            const annualTotal = monthly * 11;
            el.textContent = `Pay $${annualTotal}/yr (1 month free)`;
          }
        }
      } else {
        el.textContent = '';
      }
    });
  }

  // ---- FAQ Accordion ----
  function toggleFaq(button) {
    const item = button.closest('.forterra-pricing .faq-item');
    const wasOpen = item.classList.contains('open');

    // Close all
    document.querySelectorAll('.forterra-pricing .faq-item.open').forEach(el => {
      el.classList.remove('open');
    });

    // Toggle clicked
    if (!wasOpen) {
      item.classList.add('open');
    }
  }

  function renderComparison() {
    const plans = ['basic', 'standard', 'premium', 'sentricon', 'mosquito'];
    const wrap = document.getElementById('compareTableWrap');
    if (!wrap) return;
    
    const totalCols = plans.length + 1;

    let html = '<table class="compare-table"><thead><tr>';
    html += '<th style="text-align:left;">Feature</th>';
    plans.forEach(key => {
      const p = PLAN_DATA[key];
      const cls = p.recommended ? ' class="recommended-col"' : '';
      html += '<th' + cls + '>' + p.name + '</th>';
    });
    html += '</tr></thead><tbody>';

    COMPARE_FEATURES.forEach(row => {
      if (row.category) {
        html += '<tr class="category-row"><td colspan="' + totalCols + '">' + row.category + '</td></tr>';
      } else {
        html += '<tr><td>' + row.label + '</td>';
        plans.forEach(key => {
          html += row[key]
            ? '<td><span class="check">&#10003;</span></td>'
            : '<td><span class="dash">&mdash;</span></td>';
        });
        html += '</tr>';
      }
    });

    html += '</tbody></table>';
    wrap.innerHTML = html;
  }
